ACTOR PigShell
{
  Radius 4
  Height 6
  Speed 30
  damage (random(50,100))
  Projectile
  Scale 0.4
  Gravity 0.65
  -NOGRAVITY
  +DEHEXPLOSION
  +RANDOMIZE
  +FORCERADIUSDMG
  +BOUNCEONWALLS
  +BOUNCEONFLOORS
  +BOUNCEONACTORS
  +ROCKETTRAIL
  ReactionTime 2
  RenderStyle Add
  Alpha 1
  SeeSound ""
  DeathSound "Primo/Hit"
  BounceSound "Nade/Bounce"
  Decal "Scorch"
  DamageType Explosion
  States
  {
  Spawn:
    FLAR GHB 1 NoDelay 
    FLAR CDE 1  
	FLAR F 1 A_CountDown
    Loop
  Death:
  	TNT1 A 0 A_PlaySound("Distant/MedExp",3)
    TNT1 A 0 A_SpawnDebris("BoltSpark")
	SMEX A 1 BRIGHT A_Scream
	SMEX B 1 BRIGHT A_Explode(50,96)
	SMEX CDEFGHIJKL 2 BRIGHT
	Stop
  }
}
