	CROW A 0 A_CustomPunch(random(3, 6) * 5, TRUE, 0, "CrowbarPuff", 90)
	CROW VWXY 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 5 a_refire("upswing")
	SOLR EDC 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	goto ready
   Upswing:
    TNT1 A 4
    CRO2 ABC 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	CRO2 A 0 A_CustomPunch(random(3, 6) * 5, TRUE, 0, "CrowbarPuff", 90)
	CRO2 DEFGH 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)

	SOLR EDC 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	goto ready
 Deselect:
  SOLR A 0 A_LIGHT0
  SOLR A 0 A_takeinventory("Lightlit",1)
  SOLR A 0 A_lower
  SOLR A 1 A_Lower
  Loop
 Select:
  SOLR A 1 A_Raise
  SOLR A 0 A_Raise
  Loop
 Fire:
  TNT1 A 0 a_JumpIfInventory("lightlit",1,"unlight")
  Goto light
 Unlight:
  SOLR A 0 A_LIGHT0
  SOLR A 0 A_PlayWeaponSound("weapons/lampclickoff")
  SOLR A 1 A_takeinventory("lightlit",1)
  goto ready
 light:
  SOLR A 0 A_PlayWeaponSound("weapons/lampclickon")
  SOLR A 1 A_giveinventory("lightlit",1)
  goto ready
 }
}
