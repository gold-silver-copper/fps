ACTOR SawedOffSelector : actor replaces SuperShotgun
{
States
	{
	Spawn:
		TNT1 A 0 NoDelay A_JumpIf(GetCVar("SawedChoice") == 0, "Weapon1")
		TNT1 A 0 A_JumpIf(GetCVar("SawedChoice") == 1, "Weapon2")
		TNT1 A 0 A_JumpIf(GetCVar("SawedChoice") == 2, "Weapon3")
		Stop
    Weapon1:
		TNT1 A 0 A_jump(96,"Weapon3")
	Weapon2:
		TNT1 A 0 A_SpawnItem("ShortySpawn")
		Stop
	Weapon3:
		TNT1 A 0 A_SpawnItem("HunterSpawn")
		Stop
	}
}

ACTOR SawedOff : Weapon

  Weapon.Kickback 150

  Weapon.AmmoUse1 1
  Weapon.AmmoUse2 1

  Weapon.AmmoType2 "shotgunammo"
  Weapon.AmmoType1 "shotgunLoaded"

  Weapon.BobSpeed 1.5
  Weapon.BobRangeX .2
  Weapon.BobRangeY .5
  Weapon.Kickback 120
  Obituary "%o chewed on %k's boomstick."
  Tag "Sawed-off Shotgun"
  weapon.slotnumber 3
  scale 0.6
  weapon.upsound "weapons/sawedoffdeploy"
  +AMMO_CHECKBOTH
  +NOALERT
  +weapon.meleeweapon
  Inventory.Pickupmessage "Got the Sawed-off shotgun!"
  Inventory.PickupSound "items/gunspickup"

  Ready:

  Deselect:

	SAWN A 0 A_Lower

  Select: 
     TNT1 A 0 A_takeInventory("flashlit")
    SAWN A 0 a_JumpIfInventory("ShotgunLoaded",3,"CheetReset")
    SAWN A 1 A_Raise
	SAWN A 0 A_Raise
    Loop
  CheetReset:
    SAWN A 1 A_Raise
	SAWN A 0 A_takeinventory("Shotgunloaded",3)
	SAWN A 0 A_giveinventory("ShotgunLoaded",2)
	SAWN A 0 A_Raise
	Loop
  Fire:
    TNT1 A 0 a_JumpIfInventory("Shotgunloaded",1,"FireReal")
    Goto Reload
  FireReal:

	TNT1 A 0 A_FireBullets (7, 7, 6, 12, "hitpuff", FBF_NORANDOM)
	TNT1 A 0 a_JumpIfInventory("ShotgunLoaded",2,10)
	TNT1 A 0 A_TakeInventory("shotgunloaded",1,TIF_NOTAKEINFINITE)
    SAWF A 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)

	TNT1 A 0 A_SetPitch(Pitch-1)
	TNT1 A 0 A_FireCustomMissile("GunSmokeSpawner",0,0,8,0)
    SAWF B 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light1
	TNT1 A 0 A_SetPitch(Pitch-1)
	TNT1 A 0 A_jump(256, 5)
	TNT1 A 0 A_TakeInventory("shotgunloaded",1,TIF_NOTAKEINFINITE)
	SAWF C 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light2
	TNT1 A 0 A_SetPitch(Pitch-1)
    SAWF D 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light1
	TNT1 A 0 A_SetPitch(Pitch-1)
	SAWF E 3 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light0
	TNT1 A 0 A_SetPitch(Pitch+0.5)
	TNT1 A 0 A_AlertMonsters
	SAWF F 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+0.5)
	SAWF G 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1)
	SAWN A 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	SAWN A 0 A_checkreload
	Goto Ready
  Reload:
    SAWN A 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_jumpifinventory("Shotgunammo", 1, 1)
	goto ready
    TNT1 A 0 A_JumpIfInventory("shotgunLoaded",2,"Ready")
	TNT1 A 0 A_JumpIfInventory("shotgunloaded",1,"Work2")
 Work1:
    TNT1 A 0 A_JumpIfInventory("shotgunChambered",2,"Work3")
     SAWR AABBBB 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffopen", CHAN_7)
	SAWS CD 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
    SAWS EEFFG 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_FireCustomMissile("shellCasingSpawner",0,0,15,-3)
	TNT1 A 8
	SAWR H 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
    SAWR IJK 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffload")
	SAWR L 5 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	SAWR M 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	goto ReloadLoop
Work2:
    SAWR AABBBB 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffopen", CHAN_7)
	SAWS CD 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
    SAWH EEFFG 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_FireCustomMissile("shellCasingSpawner",0,0,15,-3)
	TNT1 A 8
	SAWH H 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)

	SAWR K 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffload")
	SAWR L 5 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	SAWR M 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)

	SAWR CD 1 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)

	SAWR H 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
    SAWR IJK 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffload")
	SAWR L 5 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
	SAWR M 2 A_WeaponReady(WRF_NOFIRE|WRF_ALLOWUSER1)
  ReloadLoop:
	TNT1 A 0 A_Giveinventory("Shotgunloaded",1)
	TNT1 A 0 A_Giveinventory("Shotgunchambered",1)
	TNT1 A 0 A_TakeInventory("shotgunammo",1,TIF_NOTAKEINFINITE)
	TNT1 A 0 a_JumpIfInventory("shotgunloaded",2,"Reloaddone2")
	TNT1 A 0 a_JumpIfInventory("shotgunammo",1,"Reloadloop")
    Goto ReloadDone1
  ReloadDone1:
    TNT1 A 4
    SAWS N 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)
	SAWS DC 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)
	TNT1 A 0 A_PlaySound ("weapons/sawedoffclose")
    SAWR A 3 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)
	SAWF EFG 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)
	SAWN A 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)

  ReloadDone2:
    TNT1 A 4
    SAWR N 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)
	SAWR DC 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH|WRF_ALLOWUSER1)

    TNT1 A 0 a_JumpIfInventory("Shotgunloaded",2,"FireDouble")
	TNT1 A 0 a_JumpIfInventory("Shotgunloaded",1,"FireReal")

	TNT1 A 0 A_FireBullets (7, 7, 6, 14, "hitpuff", FBF_NORANDOM)
	SAWF A 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light2
	TNT1 A 0 A_SetPitch(Pitch-2)
	SAWD A 1 BRIGHT A_WEAPONREADY(WRF_NOFIRE|WRF_NOSWITCH)
	 TNT1 A 0 A_PlayWeaponSound ("weapons/sawedoffdouble")
	TNT1 A 0 A_FireBullets (7, 7, 6, 16, "hitpuff", FBF_NORANDOM)
	TNT1 A 0 A_TakeInventory("shotgunLoaded",2,TIF_NOTAKEINFINITE)
    SAWD A 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_Light(3)
	TNT1 A 0 A_SetPitch(Pitch-4)
	TNT1 A 0 A_FireCustomMissile("GunSmokeSpawner",0,0,8,0)
    SAWF B 2 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_FireCustomMissile("GunSmokeSpawner",0,0,8,0)
	TNT1 A 0 A_Light1
	TNT1 A 0 A_SetPitch(Pitch-3)
	SAWD B 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1)
	SAWD C 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1)
	SAWD C 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1.5)
	TNT1 A 0 A_Light0
	TNT1 A 0 A_AlertMonsters
	SAWF F 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1.5)
	SAWF G 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+2)
	SAWN A 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	TNT1 A 0 A_SetPitch(Pitch+1)
	SAWN A 0 A_checkreload
	Goto ready
   User1:	
	    SAWN ABCD 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A    3 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
        KNIF ABCD 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
  	    TNT1 A 0 A_CustomPunch(random(5, 10) * 5, TRUE, 0, "knifePuff", 80)
	    KNIF EFG 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 10 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		SAWN DCBA 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		goto ready
   Zoom:
	    SAWN A 0  A_giveinventory("lightlit",1)
	    SAWN A 0  A_SelectWeapon("crowbar")
		SAWN A 0  A_SelectWeapon("crowbar2")
		goto deselect
    Spawn:
		SAWP A 10
		SAWP A 1 bright
		loop
  }
